/** getting shift */
function getShift(_exp1, _exp2){
		
    var x = 0;
    
    if(_exp1 > _exp2){
    		x = 1;
    }
    else if(_exp1 < _exp2){
    		x = -1;
    }
    
		return x*Math.abs(_exp1 - _exp2);
}
